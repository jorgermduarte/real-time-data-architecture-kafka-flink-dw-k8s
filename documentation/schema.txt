Table D_STUDENTS {
  STUDENT_ID int [pk, increment]
  SOCIOECONOMIC_ID int
  DEMOGRAPHIC_ID int
  NAME varchar(100) [not null]
}

  STUDENT_DEMOGRAPHIC_ID int [pk, increment]

  GENDER varchar(20)
  ETHNICITY varchar(50)
  CITY_OF_BIRTH varchar(100)
  COUNTRY_OF_BIRTH varchar(100)
  CURRENT_RESIDENCE_TYPE varchar(50)
}

Table D_SOCIOECONOMIC_DATA {
  SOCIOECONOMIC_ID int [pk, increment]

  FAMILY_INCOME decimal(10, 2)
  INCOME decimal(10, 2)
  RESPONSABLE_PARENT_EDUCATION_LEVEL varchar(100)
  RESPONSABLE_PARENT_OCCUPATION varchar(100)
  HAS_INTERNET_ACCESS boolean
  HAS_COMPUTER_ACCESS boolean
  WORKING_STATUS varchar(50)
}

  COURSE_ID int [pk, increment]

  DURATION_YEARS int
}

Table D_FIELDS_OF_STUDY {
  FIELD_ID int [pk, increment]
  FIELD_NAME varchar(50)
}

Table D_SUBJECTS {
  SUBJECT_ID int [pk, increment]
  SUBJECT_NAME varchar(100)
  COURSE_ID int
  ECTS int
  SUBJECT_TYPE varchar(50)
  SEMESTER int
  YEAR int
}

Table D_TIME {
  TIME_ID int [pk, increment]
  DAY int
  MONTH int
  YEAR int
  SEMESTER int
  WEEKDAY varchar(20)
  DATE date
}

Table D_ACADEMIC_YEAR {
  ACADEMIC_YEAR_ID int [pk, increment]

  START_DATE date
  END_DATE date
}


Table D_ENROLLMENTS {
  ENROLLMENT_ID int [pk, increment]

  ENROLLMENT_MODE varchar(50)
  ENROLLMENT_DATE date
  ENROLLMENT_STATUS varchar(50)
  TUITION_FEES decimal(10,2)
}

  FINANCIAL_STATUS_ID int [pk, increment]

  TOTAL_PENDING decimal(10,2)
  FINANCIAL_SUPPORT_AMOUNT decimal(10,2)
  STATUS varchar(20)
